# Plan 09-02: Element-to-Email Renderers + tiptapToHtml Converter

## Goal

Implement the two lowest-level modules of the export pipeline: `tiptapToHtml.ts` (converts
TipTap JSON → HTML string, no TipTap deps) and `elementRenderers.tsx` (converts each of the
5 `ElementUnion` types to a `React.ReactElement` suitable for react-email rendering).
Turns the 10 RED tests in `tiptapToHtml.test.ts` and `elementRenderers.test.ts` GREEN.

## Context

The TipTap JSON format produced by Phase 7 (`StarterKit + TextStyleKit + TextAlign`) is well-
defined. The bespoke converter avoids any TipTap server-side dependencies — it handles only
the node/mark types that Phase 7 actually emits. Element renderers produce React elements
(not HTML strings) because `documentToEmailTree` (Plan 09-03) composes them into a react-email
component tree that is later passed to `render()`.

## Tasks

### Task 1: Create `tiptapToHtml.ts` — Bespoke TipTap JSON → HTML Converter

Create `apps/server/src/export/tiptapToHtml.ts`:

```typescript
// apps/server/src/export/tiptapToHtml.ts
// Bespoke converter: TipTap JSON → HTML string, with no TipTap runtime dependencies.
// Handles the exact node/mark types emitted by Phase 7's RICH_TEXT_EXTENSIONS:
//   StarterKit (paragraph, heading, text, bulletList, orderedList, listItem, hardBreak, doc)
//   TextStyleKit (textStyle mark with color/fontSize/fontFamily attrs)
//   TextAlign extension (textAlign attr on paragraph and heading blocks)
//
// IMPORTANT: Do NOT import from @tiptap/* — server has no TipTap packages installed.

// ── Type mirrors (from apps/client/src/types/newsletter.ts) ──────────────────
// Duplicated to keep server standalone. Keep in sync with client types.

interface TiptapNode {
  type: string;
  attrs?: Record<string, unknown>;
  content?: TiptapNode[];
  marks?: Array<{ type: string; attrs?: Record<string, unknown> }>;
  text?: string;
}

interface TiptapJSONDoc {
  type: 'doc';
  content: TiptapNode[];
}

// PRESET_STYLES mirrors apps/client/src/components/builder/RichTextStaticRenderer.tsx
// and apps/client/src/lib/tiptap-extensions.ts PRESET_STYLES constant.
// MUST be kept in sync manually.
const PRESET_STYLES: Record<string, { fontSize: string; fontWeight: string; lineHeight: string; fontFamily?: string }> = {
  header:    { fontSize: '28px', fontWeight: '700', lineHeight: '1.2' },
  subheader: { fontSize: '20px', fontWeight: '600', lineHeight: '1.3' },
  body:      { fontSize: '16px', fontWeight: '400', lineHeight: '1.6' },
  code:      { fontSize: '14px', fontWeight: '400', lineHeight: '1.5', fontFamily: 'monospace' },
};

// ── Internal helpers ─────────────────────────────────────────────────────────

/** Convert a textAlign attr value to an inline style string fragment. */
function textAlignStyle(attrs?: Record<string, unknown>): string {
  const align = attrs?.textAlign;
  if (align && typeof align === 'string' && align !== 'left') {
    return `text-align:${align};`;
  }
  return '';
}

/** Wrap text in mark tags. Applied innermost first (last mark in array = outermost tag). */
function applyMarks(text: string, marks?: Array<{ type: string; attrs?: Record<string, unknown> }>): string {
  if (!marks || marks.length === 0) return text;

  let result = text;
  // Apply marks in reverse order so the first mark ends up innermost
  for (let i = marks.length - 1; i >= 0; i--) {
    const mark = marks[i];
    if (!mark) continue;
    switch (mark.type) {
      case 'bold':
        result = `<strong>${result}</strong>`;
        break;
      case 'italic':
        result = `<em>${result}</em>`;
        break;
      case 'underline':
        result = `<u style="text-decoration:underline;">${result}</u>`;
        break;
      case 'link': {
        const href = mark.attrs?.href ?? '#';
        const target = mark.attrs?.target ?? '_blank';
        result = `<a href="${href}" target="${String(target)}" style="color:inherit;">${result}</a>`;
        break;
      }
      case 'textStyle': {
        // textStyle mark: emits color, fontSize, fontFamily from TextStyleKit
        const parts: string[] = [];
        if (mark.attrs?.color)      parts.push(`color:${String(mark.attrs.color)};`);
        if (mark.attrs?.fontSize)   parts.push(`font-size:${String(mark.attrs.fontSize)};`);
        if (mark.attrs?.fontFamily) parts.push(`font-family:${String(mark.attrs.fontFamily)};`);
        if (parts.length > 0) {
          result = `<span style="${parts.join('')}">${result}</span>`;
        }
        break;
      }
      // Unknown mark types: pass through unchanged
    }
  }
  return result;
}

/** Recursively convert a TipTap node (or array of nodes) to an HTML string. */
function convertNode(node: TiptapNode): string {
  switch (node.type) {
    case 'doc':
      return (node.content ?? []).map(convertNode).join('');

    case 'paragraph': {
      const inner = (node.content ?? []).map(convertNode).join('');
      const align = textAlignStyle(node.attrs);
      const style = align ? ` style="${align}"` : '';
      return `<p${style}>${inner}</p>`;
    }

    case 'heading': {
      const level = (node.attrs?.level as number) ?? 1;
      const inner = (node.content ?? []).map(convertNode).join('');
      const align = textAlignStyle(node.attrs);
      const style = align ? ` style="${align}"` : '';
      return `<h${level}${style}>${inner}</h${level}>`;
    }

    case 'text': {
      // Escape HTML entities in raw text
      const escaped = (node.text ?? '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;');
      return applyMarks(escaped, node.marks);
    }

    case 'hardBreak':
      return '<br/>';

    case 'bulletList':
      return `<ul>${(node.content ?? []).map(convertNode).join('')}</ul>`;

    case 'orderedList':
      return `<ol>${(node.content ?? []).map(convertNode).join('')}</ol>`;

    case 'listItem':
      // listItem wraps a paragraph; strip the outer <p> tags for cleaner list rendering
      return `<li>${(node.content ?? []).map((child) => {
        if (child.type === 'paragraph') {
          return (child.content ?? []).map(convertNode).join('');
        }
        return convertNode(child);
      }).join('')}</li>`;

    default:
      // Unknown node types: recurse into children if present, otherwise ignore
      if (node.content) return node.content.map(convertNode).join('');
      return '';
  }
}

// ── Public API ───────────────────────────────────────────────────────────────

/**
 * Convert a TipTap JSON document to an HTML string.
 *
 * @param doc       The TipTap document: { type: 'doc', content: TiptapNode[] }
 * @param textStyle Optional preset key ('header'|'subheader'|'body'|'code').
 *                  When provided, wraps output in a <div> with PRESET_STYLES applied.
 * @returns         HTML string. Safe for use with dangerouslySetInnerHTML.
 */
export function tiptapToHtml(
  doc: TiptapJSONDoc | { type: string; content?: TiptapNode[] },
  textStyle?: 'header' | 'subheader' | 'body' | 'code',
): string {
  // Convert the doc node directly (handles type: 'doc' as well as any root node)
  const innerHtml = convertNode(doc as TiptapNode);

  if (!textStyle) return innerHtml;

  const preset = PRESET_STYLES[textStyle];
  if (!preset) return innerHtml;

  // Build inline style from preset
  const styleParts: string[] = [
    `font-size:${preset.fontSize};`,
    `font-weight:${preset.fontWeight};`,
    `line-height:${preset.lineHeight};`,
  ];
  if (preset.fontFamily) {
    styleParts.push(`font-family:${preset.fontFamily};`);
  }

  return `<div style="${styleParts.join('')}">${innerHtml}</div>`;
}
```

**Key implementation rules:**
- NO `display:flex` or `display:grid` anywhere in the output (EXPORT-02)
- HTML entity escaping on text nodes (`&`, `<`, `>`) prevents XSS in exported files
- `listItem` strips inner `<p>` tags for valid `<li>` markup
- Unknown node/mark types silently pass through (forward-compatibility)
- `textStyle` wrapper uses only a `<div>` — not a `<table>` — because react-email wraps it

### Task 2: Create `elementRenderers.tsx` — ElementUnion → React.ReactElement

Create `apps/server/src/export/elementRenderers.tsx`:

```tsx
// apps/server/src/export/elementRenderers.tsx
// Maps each ElementUnion type to a React.ReactElement for use in the react-email tree.
//
// Rules:
//   - No display:flex, display:grid (EXPORT-02)
//   - No CSS classes — all styles inline (juice only inlines existing inline styles + <style> tags;
//     since react-email renders with inline styles, this is already satisfied)
//   - Images: explicit width + height="auto" attrs + display:block style
//   - Buttons: <a> anchor elements (not <button>) for email client compatibility
//   - Divider: <hr> with border shorthand split into borderTop (email-safe)
import React from 'react';
import { tiptapToHtml } from './tiptapToHtml.js';

// ── Type mirrors ─────────────────────────────────────────────────────────────
// Duplicated from apps/client/src/types/newsletter.ts — server is standalone.

interface ImageElement {
  type: 'image';
  id: string;
  src: string;
  alt: string;
  width?: string;
}

interface ImageLinkElement {
  type: 'image-link';
  id: string;
  src: string;
  alt: string;
  href: string;
  width?: string;
}

interface ButtonElement {
  type: 'button';
  id: string;
  label: string;
  href: string;
  backgroundColor: string;
  textColor: string;
  borderRadius?: string;
  style: 'solid' | 'outline' | 'ghost';
}

interface RichTextElement {
  type: 'rich-text';
  id: string;
  content: { type: 'doc'; content: unknown[] };
  textStyle: 'header' | 'subheader' | 'body' | 'code';
}

interface DividerElement {
  type: 'divider';
  id: string;
  color: string;
  spacing: number;
  thickness: number;
}

// ── Width parsing helper ─────────────────────────────────────────────────────

/**
 * Parse an element width string (e.g. "300px", "100%", undefined) to a pixel number.
 * "100%" or undefined → defaults to 600 (full content width).
 * "300px" → 300.
 */
function parseWidthPx(width?: string, fallback = 600): number {
  if (!width) return fallback;
  if (width.endsWith('%')) return fallback;
  const n = parseInt(width, 10);
  return isNaN(n) ? fallback : n;
}

// ── Renderers ────────────────────────────────────────────────────────────────

/** Render an ImageElement as an email-safe <img>. */
export function imageToEmail(el: ImageElement): React.ReactElement {
  const widthPx = parseWidthPx(el.width, 600);
  return (
    <img
      src={el.src}
      alt={el.alt}
      width={widthPx}
      style={{
        maxWidth: '100%',
        height: 'auto',
        display: 'block',
        border: '0',
        outline: 'none',
        textDecoration: 'none',
      }}
    />
  );
}

/** Render an ImageLinkElement as an <a> wrapping an <img>. */
export function imageLinkToEmail(el: ImageLinkElement): React.ReactElement {
  const widthPx = parseWidthPx(el.width, 600);
  return (
    <a
      href={el.href}
      target="_blank"
      rel="noopener noreferrer"
      style={{ display: 'block', textDecoration: 'none' }}
    >
      <img
        src={el.src}
        alt={el.alt}
        width={widthPx}
        style={{
          maxWidth: '100%',
          height: 'auto',
          display: 'block',
          border: '0',
          outline: 'none',
          textDecoration: 'none',
        }}
      />
    </a>
  );
}

/**
 * Render a ButtonElement as an email-safe <a> anchor.
 * Uses <a> (not <button>) for email client compatibility.
 *
 * Variants:
 *   solid   — filled background, text color from textColor
 *   outline — transparent background, colored border and text (color = backgroundColor)
 *   ghost   — no background, no border, link color = backgroundColor
 */
export function buttonToEmail(el: ButtonElement): React.ReactElement {
  const baseStyle: React.CSSProperties = {
    display: 'inline-block',
    padding: '12px 24px',
    fontSize: '16px',
    fontWeight: '600',
    textDecoration: 'none',
    textAlign: 'center',
    borderRadius: el.borderRadius ?? '4px',
    cursor: 'pointer',
    // Prevent font size scaling in some email clients
    WebkitTextSizeAdjust: 'none',
    msTextSizeAdjust: 'none',
  };

  let variantStyle: React.CSSProperties;

  switch (el.style) {
    case 'outline':
      variantStyle = {
        backgroundColor: 'transparent',
        color: el.backgroundColor,
        border: `2px solid ${el.backgroundColor}`,
      };
      break;
    case 'ghost':
      variantStyle = {
        backgroundColor: 'transparent',
        color: el.backgroundColor,
        border: 'none',
      };
      break;
    case 'solid':
    default:
      variantStyle = {
        backgroundColor: el.backgroundColor,
        color: el.textColor,
        border: 'none',
      };
      break;
  }

  return (
    <a
      href={el.href}
      target="_blank"
      rel="noopener noreferrer"
      style={{ ...baseStyle, ...variantStyle }}
    >
      {el.label}
    </a>
  );
}

/**
 * Render a RichTextElement.
 * Calls tiptapToHtml() to convert TipTap JSON → HTML string, then injects via
 * dangerouslySetInnerHTML. This is safe because tiptapToHtml HTML-escapes text nodes.
 */
export function richTextToEmail(el: RichTextElement): React.ReactElement {
  const html = tiptapToHtml(
    el.content as Parameters<typeof tiptapToHtml>[0],
    el.textStyle,
  );
  return (
    <div
      dangerouslySetInnerHTML={{ __html: html }}
      style={{ margin: '0', padding: '0' }}
    />
  );
}

/** Render a DividerElement as an email-safe <hr>. */
export function dividerToEmail(el: DividerElement): React.ReactElement {
  return (
    <hr
      style={{
        border: 'none',
        borderTop: `${el.thickness}px solid ${el.color}`,
        margin: `${el.spacing}px 0`,
        padding: '0',
        height: '0',
        // Explicit height:0 + border-top pattern is most email-client compatible
      }}
    />
  );
}
```

**Implementation rules:**
1. **No `<button>` elements** — email clients strip or break interactive elements. Use `<a>` for buttons.
2. **No `display:flex`** — use `display:inline-block` or `display:block` only.
3. **`display:block` on `<img>`** — prevents bottom whitespace gap in Outlook/Gmail.
4. **`border: '0'` on `<img>`** — removes default image borders in older Outlook.
5. **`tiptapToHtml` call in `richTextToEmail`** — pass `el.content` as the first arg and `el.textStyle` as second. The converter wraps output in a preset-styled `<div>` when textStyle is provided.
6. **HR element for divider** — `border: 'none'` clears the default `<hr>` border, then `borderTop` is set explicitly. This is the standard email-safe divider pattern.
