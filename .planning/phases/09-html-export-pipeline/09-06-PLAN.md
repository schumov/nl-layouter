# Plan 09-06: Export API Route

## Goal

Implement the Fastify route `POST /newsletters/:id/export` that wires the complete export
pipeline to HTTP. Loads newsletter + preset HTML from DB, calls `renderToEmailHtml()`, and
returns the HTML file as a download attachment. Gracefully handles missing presets (ECONNRESET
or absent seed data) by falling back to empty string header/footer HTML.

## Context

The server runs on `http://localhost:3001`. The client calls it directly with
`http://localhost:3001/newsletters/${id}/export` (no `/api` prefix — confirmed by examining
existing hooks in `apps/client/src/hooks/`). The route follows the same Fastify plugin pattern
as `apps/server/src/routes/newsletters.ts`.

**DB access pattern** (from `newsletters.ts`):
```typescript
import { eq } from 'drizzle-orm';
import { db } from '../db/connection.js';
import { newsletters, presets } from '../db/schema.js';
const [row] = await db.select().from(newsletters).where(eq(newsletters.id, id));
```

## Tasks

### Task 1: Create `apps/server/src/routes/export.ts`

```typescript
// apps/server/src/routes/export.ts
// Fastify plugin: POST /newsletters/:id/export
// Loads newsletter doc + preset HTML from DB, runs the export pipeline,
// returns a self-contained .html file as a download attachment (EXPORT-01, EXPORT-07).
//
// Pattern mirrors apps/server/src/routes/newsletters.ts
import type { FastifyPluginAsync } from 'fastify';
import { eq } from 'drizzle-orm';
import { db } from '../db/connection.js';
import { newsletters, presets } from '../db/schema.js';
import { renderToEmailHtml } from '../export/pipeline.js';

// Local opaque type — avoid cross-package import (same pattern as newsletters.ts)
type NewsletterDoc = {
  header?:      { presetId: string; variables: Record<string, string> };
  footer?:      { presetId: string; variables: Record<string, string> };
  rows?:        unknown[];
  globalStyles?: {
    fontFamily:      string;
    backgroundColor: string;
    contentWidth:    number;
    primaryColor:    string;
  };
  preHeader?: string;
};

// ── Helper: load preset HTML from DB ────────────────────────────────────────

/**
 * Fetch the HTML content for a preset by ID.
 * Returns empty string gracefully if:
 *   - presetId is empty or undefined
 *   - preset not found in DB
 *   - DB throws (e.g. ECONNRESET on Neon cold-start)
 *
 * Empty string is a valid fallback — the export pipeline renders without header/footer.
 */
async function getPresetHtml(presetId: string | undefined): Promise<string> {
  if (!presetId || presetId.trim() === '') return '';
  try {
    const [preset] = await db
      .select({ htmlContent: presets.htmlContent })
      .from(presets)
      .where(eq(presets.id, presetId));
    return preset?.htmlContent ?? '';
  } catch {
    // Graceful fallback — ECONNRESET, Neon cold-start, or preset table missing
    // Export still proceeds; header/footer will be absent from output
    return '';
  }
}

// ── Fastify plugin ────────────────────────────────────────────────────────────

const exportRoute: FastifyPluginAsync = async (fastify) => {

  fastify.post('/newsletters/:id/export', async (request, reply) => {
    const { id } = request.params as { id: string };

    // ── 1. Load newsletter ──────────────────────────────────────────────────
    const [newsletter] = await db
      .select()
      .from(newsletters)
      .where(eq(newsletters.id, id));

    if (!newsletter) {
      return reply.code(404).send({ error: 'Newsletter not found' });
    }

    const doc = newsletter.document as NewsletterDoc;

    // ── 2. Load preset HTML (graceful fallback on missing/error) ────────────
    const [headerHtml, footerHtml] = await Promise.all([
      getPresetHtml(doc.header?.presetId),
      getPresetHtml(doc.footer?.presetId),
    ]);

    // ── 3. Run export pipeline ──────────────────────────────────────────────
    const html = await renderToEmailHtml(
      doc as never,
      headerHtml,
      footerHtml,
      newsletter.title,
    );

    // ── 4. Build download filename ──────────────────────────────────────────
    // Slugify: replace non-alphanumeric chars with hyphens, lowercase
    // e.g. "My Newsletter #1!" → "my-newsletter--1-"
    const slug = newsletter.title
      .replace(/[^a-z0-9]/gi, '-')
      .replace(/-{2,}/g, '-')     // collapse consecutive hyphens
      .replace(/^-|-$/g, '')      // strip leading/trailing hyphens
      .toLowerCase() || 'newsletter';

    const filename = `${slug}.html`;

    // ── 5. Return HTML as download attachment ───────────────────────────────
    return reply
      .header('Content-Type', 'text/html; charset=UTF-8')
      .header('Content-Disposition', `attachment; filename="${filename}"`)
      .send(html);
  });

};

export default exportRoute;
```

**Security note:** The `filename` value in `Content-Disposition` is derived from the newsletter
title stored in the database. The slugify step ensures only `[a-z0-9-]` characters appear in the
filename, preventing header injection. The `html` content is generated by our pipeline from
structured data — it is NOT a user-supplied HTML string passed directly through.

(`methods: ['GET', 'HEAD', 'POST', 'PUT', 'PATCH', 'DELETE', 'OPTIONS']`), so no CORS changes
are needed for the new POST route.
