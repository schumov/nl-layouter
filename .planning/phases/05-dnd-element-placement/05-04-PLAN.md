Custom collision detection pattern (verified from @dnd-kit/core@6.3.1 source — Finding 1):
```typescript
import type { CollisionDetection } from '@dnd-kit/core';
import { closestCenter } from '@dnd-kit/core';
import { DRAG_TYPES } from '@/dnd/types';

const customCollision: CollisionDetection = (args) => {
  const dragType = args.active.data.current?.type;
  if (dragType === DRAG_TYPES.ELEMENT_CARD) {
    // Filter to only COLUMN_SLOT droppables — data.type === DRAG_TYPES.ELEMENT_CARD
    // Prevents SortableRowBlock row from stealing collision in 1-col sections
    const slotContainers = args.droppableContainers.filter(
      (c) => c.data.current?.type === DRAG_TYPES.ELEMENT_CARD,
    );
    if (slotContainers.length === 0) return [];
    return closestCenter({ ...args, droppableContainers: slotContainers });
  }
  return closestCenter(args);  // LAYOUT_CARD + CANVAS_ROW: unfiltered standard closestCenter
};
```

```typescript
    return closestCenter({ ...args, droppableContainers: slotContainers });
  }

};
```
