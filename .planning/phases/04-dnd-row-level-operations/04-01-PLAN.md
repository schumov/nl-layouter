    ```typescript
    reorderSections: (activeId, overId) =>
      set((state) => {
        if (!state.doc) return;
        const rows = state.doc.rows;
        const activeIndex = rows.findIndex((r) => r.id === activeId);
        const overIndex  = rows.findIndex((r) => r.id === overId);
        if (activeIndex === -1 || overIndex === -1 || activeIndex === overIndex) return;
        state.doc.rows = arrayMove(rows as Section[], activeIndex, overIndex);
        // Assignment (not splice) — arrayMove returns a NEW array; Immer allows assignment here
        // Same pattern as removeSection: state.doc.rows = ...filter()
      }),

    duplicateSection: (sectionId) =>
      set((state) => {
        if (!state.doc) return;
        const index = state.doc.rows.findIndex((r) => r.id === sectionId);
        if (index === -1) return;
        const clone = structuredClone<Section>(state.doc.rows[index]);
        // Assign fresh UUIDs — required so DnD keys and slot references are unique
        clone.id = crypto.randomUUID();
        clone.slots = clone.slots.map((slot) => ({
          ...slot,
          id: crypto.randomUUID(),
          element: slot.element ? { ...slot.element, id: crypto.randomUUID() } : null,
        }));
        state.doc.rows.splice(index + 1, 0, clone);
        // .splice is in-place mutation — valid in Immer (unlike arrayMove which is assignment)
      }),
    ```

    CRITICAL implementation rules:
    - `arrayMove` RETURNS a new array — must ASSIGN with `state.doc.rows = arrayMove(...)` NOT push/splice
    - `structuredClone<Section>` — TypeScript generic form; Node 22 global available in jsdom tests
    - `crypto.randomUUID()` — browser + Node 22 global; no import needed
    - Guard `if (activeIndex === -1 || overIndex === -1 || activeIndex === overIndex) return;`
      prevents no-op arrayMove calls and covers the edge-case test
    - `state.doc.rows.splice(index + 1, 0, clone)` — splice IS valid Immer mutation (unlike arrayMove)
    - Do NOT add `import { structuredClone }` — it is a global in TypeScript lib.dom.d.ts and Node 22
