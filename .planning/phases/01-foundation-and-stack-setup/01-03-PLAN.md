```typescript
import 'dotenv/config';
import { defineConfig } from 'drizzle-kit';

export default defineConfig({

  out: './drizzle',

    url: process.env['DATABASE_URL']!,
  },
  verbose: true,
  strict: true,

```
NOTE: `import 'dotenv/config'` at the top loads `apps/server/.env` (same directory as drizzle.config.ts) before drizzle-kit reads `process.env.DATABASE_URL`.

**`apps/server/src/db/connection.ts`** — Drizzle + postgres.js singleton:
```typescript
import { drizzle } from 'drizzle-orm/postgres-js';
import postgres from 'postgres';

// postgres.js queryClient — connection pool for all DB operations
// DATABASE_URL is validated by Zod in config.ts at server startup;
// this file imports process.env directly for use by drizzle-kit CLI as well.
const queryClient = postgres(process.env['DATABASE_URL']!);

export const db = drizzle({ client: queryClient });
```
NOTE: Using `process.env['DATABASE_URL']` (bracket notation) avoids TypeScript strict mode issues with `process.env.DATABASE_URL` when `noUncheckedIndexedAccess` is enabled.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
