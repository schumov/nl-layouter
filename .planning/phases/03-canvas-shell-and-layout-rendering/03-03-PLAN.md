**Step 1 — Create `apps/client/src/components/builder/BuilderCanvas.tsx`:**
```typescript
import React from 'react';
import type { NewsletterDoc } from '../../types/newsletter';
import { RowBlock } from './RowBlock';

interface BuilderCanvasProps {
  doc: NewsletterDoc | null;
}

export function BuilderCanvas({ doc }: BuilderCanvasProps) {
  return (
    <div className="flex-[3] min-w-0 overflow-y-auto bg-canvas">
      <div className="max-w-[640px] mx-auto px-4 py-8 space-y-2">
        {doc?.rows.map((section) => (
          <RowBlock key={section.id} section={section} />
        ))}
        {(!doc || doc.rows.length === 0) && (
          <p className="text-center text-sm text-muted-foreground py-16">
            No sections yet. Drag a layout from the palette to begin.
          </p>
        )}
      </div>
    </div>
  );
}
```
Critical class requirements:
- `flex-[3]` — arbitrary Tailwind v4 flex value; produces `flex: 3`; pairs with `flex-[2]` on BuilderPalette for 60/40 split
- `min-w-0` — prevents flex child overflow; mandatory on all flex children
- `overflow-y-auto` — canvas scrolls independently; BuilderPalette does NOT need this
- `bg-canvas` — uses the `--color-canvas: #f4f4f5` token added in Plan 03-01; do NOT use `bg-neutral-100` (different hex value)
- `max-w-[640px] mx-auto px-4 py-8 space-y-2` — content area wrapper
- Empty state shown when `(!doc || doc.rows.length === 0)` — both null doc AND empty rows array trigger this
