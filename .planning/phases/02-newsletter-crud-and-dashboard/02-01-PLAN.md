```typescript
// apps/server/src/db/schema.ts
// Phase 2: newsletters table definition.
// NewsletterDoc is a TypeScript-only type annotation on the JSONB column —
// erased at runtime, safe to import cross-package.
import { pgTable, uuid, text, jsonb, timestamp } from 'drizzle-orm/pg-core';
import type { NewsletterDoc } from '../../../apps/client/src/types/newsletter.js';

export const newsletters = pgTable('newsletters', {
  id:        uuid('id').primaryKey().defaultRandom(),
  title:     text('title').notNull(),
  document:  jsonb('document').$type<NewsletterDoc>().notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull()
               .$onUpdate(() => new Date()),
});

// TypeScript helpers — import these in route handlers for type-safe query results
export type Newsletter    = typeof newsletters.$inferSelect;
export type NewNewsletter = typeof newsletters.$inferInsert;
```

CRITICAL rules:
- Import path for NewsletterDoc: `'../../../apps/client/src/types/newsletter.js'` — uses ESM `.js` suffix (server TypeScript rule)
- `updatedAt` MUST have BOTH `.defaultNow()` AND `.$onUpdate(() => new Date())` — `$onUpdate` alone leaves NULL on INSERT (Pitfall 6)
- `$type<NewsletterDoc>()` is TypeScript-only — Drizzle stores/retrieves plain JSON; this just narrows the inferred TypeScript type
- Remove the `export {}` placeholder — the new exports replace it
