```typescript
import React, { useState } from 'react';
import { useNavigate } from 'react-router';
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { useCreateNewsletter } from '../../hooks/useCreateNewsletter';

interface CreateNewsletterDialogProps {
  triggerLabel?: string;
}

export function CreateNewsletterDialog({ triggerLabel = 'New Newsletter' }: CreateNewsletterDialogProps) {
  const [open, setOpen]   = useState(false);
  const [title, setTitle] = useState('');
  const navigate          = useNavigate();
  const createMutation    = useCreateNewsletter();

  const handleSubmit = () => {
    if (!title.trim()) return;
    createMutation.mutate(title.trim(), {
      onSuccess: (data) => {
        // D-06: navigate directly to builder — dialog unmounts with navigation
        navigate(`/newsletters/${data.id}`);
      },
      // onError: leave dialog open (Pitfall 5); isError shows feedback below
    });
  };

  const handleOpenChange = (nextOpen: boolean) => {
    if (!nextOpen) {
      // Reset state when dialog closes (Cancel or overlay click)
      setTitle('');
      createMutation.reset();
    }
    setOpen(nextOpen);
  };

  return (
    <Dialog open={open} onOpenChange={handleOpenChange}>
      <DialogTrigger asChild>
        <Button>{triggerLabel}</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>New Newsletter</DialogTitle>
          <DialogDescription>Give your newsletter a name to get started.</DialogDescription>
        </DialogHeader>

        <div className="flex flex-col gap-4 pt-2">
          <Input
            autoFocus
            placeholder="Newsletter name"
            value={title}
            onChange={(e) => setTitle(e.target.value)}
            onKeyDown={(e) => { if (e.key === 'Enter') handleSubmit(); }}
            disabled={createMutation.isPending}
          />

          {createMutation.isError && (
            <p className="text-sm text-destructive">
              Failed to create newsletter. Please try again.
            </p>
          )}

          <div className="flex justify-end gap-2">
            <Button
              variant="outline"
              onClick={() => handleOpenChange(false)}
              disabled={createMutation.isPending}
            >
              Discard
            </Button>
            <Button
              onClick={handleSubmit}
              disabled={!title.trim() || createMutation.isPending}
            >
              {createMutation.isPending ? 'Creating…' : 'Create Newsletter'}
            </Button>
          </div>
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

Create `apps/client/src/pages/BuilderPage.tsx`.

This page:
1. Reads `:id` from route params via `useParams<{ id: string }>()`
2. Fetches full newsletter via `useNewsletter(id!)`
3. Loads document into Zustand via `setDoc(data.document)` when data resolves (useEffect)
4. Clears Zustand doc on unmount via cleanup function `() => clearDoc()`
5. Calls `useAutoSave(id!)` for the save status
6. Passes `id`, `title`, and `saveStatus` to `BuilderHeader` (imported from Plan 08 — add placeholder for now)
7. Returns `null` during initial load (D-10: no loading indicator on builder)

```typescript
import React, { useEffect } from 'react';
import { useParams } from 'react-router';
import { useNewsletter } from '../hooks/useNewsletters';
import { useNewsletterStore } from '../store/useNewsletterStore';
import { useAutoSave } from '../hooks/useAutoSave';

// BuilderHeader will be created in Plan 08 — import it here as a forward reference
// If Plan 07 and Plan 08 execute sequentially, this import resolves correctly.
// If Plan 07 executes before Plan 08 exists, use a temporary placeholder:
//   const BuilderHeader = () => <div className="h-14 border-b" />;
import BuilderHeader from '../components/builder/BuilderHeader';

export default function BuilderPage() {
  const { id }                       = useParams<{ id: string }>();
  const { data, isPending, isError } = useNewsletter(id!);
  const { setDoc, clearDoc }         = useNewsletterStore();
  const { saveStatus }               = useAutoSave(id!);

  // Load doc into Zustand when query resolves; clear on unmount (navigate-away)
  useEffect(() => {
    if (data) setDoc(data.document);
    return () => clearDoc();
  }, [data, setDoc, clearDoc]);

  // D-10: No loading indicator during initial load — return null (not spinner)
  if (isPending) return null;

  if (isError) {
    return (
      <div className="flex items-center justify-center h-screen text-sm text-destructive p-4">
        Error loading newsletter. It may have been deleted.
      </div>
    );
  }

  return (
    <div className="flex flex-col h-screen">
      <BuilderHeader
        id={id!}
        title={data?.title ?? ''}
        saveStatus={saveStatus}
      />
      {/* Canvas placeholder — Phase 3 fills this */}
      <main className="flex-1 bg-neutral-100" />
    </div>
  );
}
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-07-01 | Tampering | CreateDialog title input | mitigate | Client-side: `title.trim()` before sending; server-side: Zod `z.string().min(1)` rejects empty strings |
| T-02-07-02 | Spoofing | useParams :id in BuilderPage | accept | URL :id is user-navigable; if ID is garbage, `useNewsletter` GET returns 404 → `isError = true` → error UI shown; no crash |
