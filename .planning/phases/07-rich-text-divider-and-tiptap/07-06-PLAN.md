```tsx
// apps/client/src/components/builder/ElementRenderer.tsx
//
// ⚠️ TAILWIND V4 RULE: All class names must be complete string literals.
// D-07/D-10: isSelected prop gates the lifecycle — only ONE RichTextEditor is mounted at a time.
//            When NOT selected, rich-text renders via RichTextStaticRenderer (no ProseMirror).
// assertNeverElement default is preserved for exhaustiveness checking.

import React from 'react';
import type { ElementUnion } from '../../types/newsletter';
import { assertNeverElement } from '../../types/newsletter';
import { ImageRenderer } from './ImageRenderer';
import { ImageLinkRenderer } from './ImageLinkRenderer';
import { ButtonRenderer } from './ButtonRenderer';
import { RichTextEditor } from './RichTextEditor';
import { RichTextStaticRenderer } from './RichTextStaticRenderer';
import { DividerRenderer } from './DividerRenderer';

// ⚠️ TAILWIND V4 RULE: All class names must be complete string literals.

interface ElementRendererProps {
  element:    ElementUnion;
  isSelected?: boolean;      // true when this slot is the selected element in the store
  onUpdate?:  (patch: Partial<ElementUnion>) => void; // passed to RichTextEditor when selected
}

// Routes each element type to its canvas renderer.
// Rich-text: lifecycle guard — RichTextEditor when selected, RichTextStaticRenderer otherwise.
// Divider: always uses DividerRenderer (no editing on canvas — only InspectorPanel editing).
export function ElementRenderer({ element, isSelected = false, onUpdate }: ElementRendererProps) {
  switch (element.type) {
    case 'image':
      return <ImageRenderer element={element} />;

    case 'image-link':
      return <ImageLinkRenderer element={element} />;

    case 'button':
      return <ButtonRenderer element={element} />;

    case 'rich-text':
      // D-07: ONE editor instance max. Conditional mount ensures the ProseMirror DOM
      // is only created when this slot is selected. All other slots use the static renderer.
      return isSelected
        ? <RichTextEditor element={element} onUpdate={onUpdate ?? (() => {})} />
        : <RichTextStaticRenderer element={element} />;

    case 'divider':
      // D-08: Divider has no canvas-level editing — only InspectorPanel editing.
      return <DividerRenderer element={element} />;

    default:
      // TypeScript exhaustiveness: if all 5 cases above are handled, this line
      // can never be reached at runtime. Adding a new type to ElementUnion
      // without handling it here will cause a compile-time error.
      return assertNeverElement(element);
  }
}
```
